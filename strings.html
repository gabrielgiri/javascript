    function getFirstThreeCharacters(a) {
        return a.substring(0, 3);
    }
    
    const firstThree1 = getFirstThreeCharacters("Hello, World!");
    const firstThree2 = getFirstThreeCharacters("1561561");
    const firstThree3 = getFirstThreeCharacters("Gabriel es mi nombre");
    console.log(firstThree1);   // Hel
    console.log(firstThree2);   // 156
    console.log(firstThree3);   // Gab

    



    function getCustomCharacters(a,b,c) {
        return a.substring(b, c);
    }
    
    const custom1 = getCustomCharacters("Hello, World!", 0, 3);
    const custom2 = getCustomCharacters("1561561", 2, 4);
    const custom3 = getCustomCharacters("Gabriel es mi nombre", 3, 12);
    console.log(custom1);   // Hel
    console.log(custom2);   // 61
    console.log(custom3);   // riel es m
