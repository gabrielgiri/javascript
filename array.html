ARRAY

JAVASCRIPT

  • push() agrega un elemento al final
  • pop() borra el último del array
  • shift() borra el primero del array
  • unshift() agrega un elemento al principio del array
⬅ Volver al menú 

var hobbies = ["Futbol", "Padel", "Running", "Games", "Gym"];
var hobbiesFemale = ["Dancing", "MakeUp", "Movies", "Dinner"];

hobbies.pop(); // elimina ultimo elemento "gym"
hobbies.push("Music") // agrega "Music" al final del array
hobbies.shift(); // elimina el primer elemento "Futbol"

console.log(hobbies);
console.log(typeof(hobbies))
console.log("Index position 2 is: " + hobbies[2]);
console.log ("hobbies.length: " + hobbies.length);

const index = hobbies.findIndex((item) => item === "Games");
const indexShort = hobbies.findIndex((item) => (item === "Padel"));

console.log("Games is on position: " + index);
console.log("Padel is on position: " + indexShort);

const editedHobbies = hobbies.map((item) => item + "!");
console.log(editedHobbies);

const mergeHobbies = [...hobbies, ...hobbiesFemale];
console.log(mergeHobbies);

function range(max) {
  var retVal = [];
  for (var i = 0; i < max; i++) {
      retVal.push(i * 2);
  }
  var ultimoValor = retVal.pop();
  console.log(ultimoValor);
  return retVal;
}

console.log(range(30));

var sizeArray = range(5);
for (var i = 0; i < sizeArray.length; i++) {
  console.log(sizeArray[i]);
}

//---- Destructuring Arrays ----//
let cartas = [1, 2, 3, 4, 5, 6, 7, 10, 11, 12];

let [card1, card2, card3] = cartas;
console.log(card1, card2, card3);

let [card1_2, card2_2, card3_2, card4, ...elResto] = cartas;
console.log(card1_2, card2_2, card3_2, card4, elResto);
