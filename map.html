const map = new Map();

map.set(1, 2);
map.set(1980, 1986);

map.set("Ciudad", "Barcelona"); // se sobrescribe
map.set("Ciudad", "Madrid");    // se sobrescribe
map.set("Ciudad", "Sevilla");   // valor final

map.set("Pais", "España");

map.set("nombre", "Gabriel");
map.set("nombre", "Gabriel");   // se sobrescribe, mismo valor

const elementos = map.entries(); // iterador, NO copia

console.log(map);        // Map completo
console.log(elementos);  // MapIterator

map.delete("nombre");    // elimina la clave
console.log(elementos);  // el iterador refleja el cambio

map.forEach((valor, clave) => {
    console.log(clave, valor);
});

map.clear();             // vacía el Map
console.log(elementos);  // iterador ya sin datos
